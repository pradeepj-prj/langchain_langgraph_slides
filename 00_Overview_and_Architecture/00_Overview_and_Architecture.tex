\begin{frame}{The Problem: Why LangChain?}

\textbf{Building LLM applications without a framework means:}

\begin{itemize}
  \item Writing custom glue code for every LLM provider (OpenAI, Anthropic, Google, \ldots)
  \item Manually managing prompt templates, output parsing, error handling
  \item Reinventing conversation memory, tool calling, retrieval patterns
  \item No standardized way to compose, test, or observe your pipelines
\end{itemize}

\vspace{0.3cm}

\begin{conceptbox}[LangChain's Value Proposition]
LangChain provides \textbf{standardized interfaces} for LLM components and \textbf{composable primitives} for chaining them together --- so you can swap providers, add tools, and build agents without rewriting your application.
\end{conceptbox}

\docref{langchain.com}
\end{frame}

% ============================================================
\begin{frame}{What LangChain Is (and Isn't)}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{tipbox}[LangChain IS]
\begin{itemize}
  \item A framework for composing LLM calls
  \item An abstraction layer over providers
  \item A toolkit for agents, RAG, tools
  \item An orchestration platform (via LangGraph)
  \item Provider-agnostic by design
\end{itemize}
\end{tipbox}
\end{column}

\begin{column}{0.48\textwidth}
\begin{warningbox}[LangChain is NOT]
\begin{itemize}
  \item An LLM itself
  \item A hosted AI service
  \item A UI framework
  \item Required for all LLM apps
  \item A replacement for understanding LLMs
\end{itemize}
\end{warningbox}
\end{column}
\end{columns}

\vspace{0.3cm}
\begin{tipbox}[When to Use LangChain]
Use LangChain when you need \textbf{multi-step LLM workflows}, \textbf{tool calling}, \textbf{retrieval}, or \textbf{agents}. For a single API call, the raw provider SDK may suffice.
\end{tipbox}
\end{frame}

\begin{frame}{The Runnable Protocol}

\begin{conceptbox}[The Universal Interface]
Every component in LangChain implements the \textbf{Runnable} protocol, which provides three methods:
\begin{itemize}
  \item \texttt{invoke(input)} --- Process a single input
  \item \texttt{batch([inputs])} --- Process multiple inputs in parallel
  \item \texttt{stream(input)} --- Stream output token by token
\end{itemize}
Plus async variants: \texttt{ainvoke()}, \texttt{abatch()}, \texttt{astream()}
\end{conceptbox}

\vspace{0.3cm}

\begin{tipbox}[Why This Matters]
Because every component shares this interface, you can \textbf{compose} them freely with the pipe operator (\texttt{|}). A prompt template, a model, and a parser all ``plug in'' the same way.
\end{tipbox}

\docref{docs.langchain.com/oss/python/concepts/runnables}
\end{frame}

% ============================================================
\begin{frame}[fragile]{LCEL: LangChain Expression Language (Preview)}

\begin{codebox}[Composing with the Pipe Operator]
\begin{lstlisting}
from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser

# Define components
prompt = ChatPromptTemplate.from_template(
    "Explain {topic} in simple terms."
)
model = ChatOpenAI(model="gpt-4o")
parser = StrOutputParser()

# Compose into a chain with |
chain = prompt | model | parser

# Use it
result = chain.invoke({"topic": "quantum computing"})
print(result)
\end{lstlisting}
\end{codebox}

\begin{frame}{What Changed: LangChain v0.x $\rightarrow$ v1.x}

\begin{warningbox}[Breaking Changes in v1.0+]
\begin{itemize}
  \item \textbf{LCEL is the standard} --- Legacy chain classes (\texttt{LLMChain}, \texttt{SequentialChain}) moved to \texttt{langchain-classic}
  \item \textbf{Provider packages split out} --- \texttt{from langchain.chat\_models} $\rightarrow$ \texttt{from langchain\_openai}
  \item \textbf{langchain-core extracted} --- Interfaces live in \texttt{langchain-core}, not \texttt{langchain}
  \item \textbf{Agents built on LangGraph} --- \texttt{create\_react\_agent} returns a LangGraph graph
  \item \textbf{Python 3.10+ required} --- Dropped 3.8/3.9 support
\end{itemize}
\end{warningbox}

\vspace{0.2cm}

\begin{tipbox}[Migration Tip]
If migrating from v0.x, start by updating imports to use provider-specific packages and replacing legacy chains with LCEL pipelines.
\end{tipbox}

\docref{docs.langchain.com/oss/python/releases/changelog}
\end{frame}

\begin{frame}{Summary}

\begin{conceptbox}[Key Takeaways]
\begin{enumerate}
  \item LangChain is a \textbf{framework for composing LLM-powered applications}, not an LLM itself
  \item The ecosystem spans \textbf{four pillars}: langchain-core (interfaces), langchain (features), LangGraph (orchestration), LangSmith (observability)
  \item Everything implements the \textbf{Runnable protocol}: \texttt{invoke()}, \texttt{batch()}, \texttt{stream()}
  \item Components compose with the \textbf{pipe operator} (\texttt{|}) via LCEL
  \item Provider packages are \textbf{separate}: install only what you need
  \item v1.x is a \textbf{major redesign} --- use LCEL, not legacy chains
\end{enumerate}
\end{conceptbox}
\end{frame}
