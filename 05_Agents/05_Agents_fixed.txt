% SLIDE 2: Mental Model - What is an Agent?
% ============================================================
\begin{frame}[fragile]
\frametitle{Mental Model: Agents as Reasoning Loops}

An \textbf{agent} is an LLM that:
\begin{enumerate}
  \item Thinks about what to do next
  \item Decides which tool(s) to call
  \item Observes the tool results
  \item Repeats until it has an answer
\end{enumerate}

\vspace{0.5cm}

Unlike chains (linear), agents are \textbf{dynamic}:
\begin{itemize}
  \item Chain: ``Always call API, then summarize''
  \item Agent: ``Analyze the query, then decide what tools I need''
\end{itemize}

\vspace{0.5cm}

\begin{conceptbox}[Key Insight]
Agents are control flow determined by the LLM, not by you.
\end{conceptbox}

\end{frame}

% ============================================================
% SLIDE 3: Formal Definition
% ============================================================
\begin{frame}
\frametitle{What is an Agent? (Formal Definition)}

\textbf{Definition:} An agent is a system that:

\vspace{0.3cm}

\begin{enumerate}
  \item Takes a user query or goal as input
  \item Uses an LLM to reason about which \textbf{tools} to invoke
  \item Calls one or more tools with appropriate arguments
  \item Observes the tool outputs
  \item Iterates until it reaches a terminal state
  \item Returns a final answer to the user
\end{enumerate}

\vspace{0.5cm}

\textbf{In LangChain v1.x:}
\begin{itemize}
  \item Agents are built on \textbf{LangGraph} under the hood
  \item \texttt{create\_react\_agent()} is the main constructor
  \item Returns a \texttt{CompiledGraph}—a runnable state machine
  \item State tracks messages, tool calls, and observations
\end{itemize}

\docref{langchain.agents}

\end{frame}

% ============================================================
% SLIDE 4: The ReAct Pattern
% ============================================================
\begin{frame}
\frametitle{The ReAct Pattern: Reason → Act → Observe}

\textbf{ReAct} = \textbf{Re}asoning + \textbf{Act}ing

\vspace{0.5cm}

\begin{center}
\begin{tabular}{ll}
\toprule
\textbf{Step} & \textbf{What Happens} \\
\midrule
Reason & LLM thinks: ``I need to do X'' \\
Act & Agent calls tool(s) based on that reasoning \\
Observe & Tool results are captured and fed back \\
Think & LLM sees results, decides next step \\
\midrule
\multicolumn{2}{c}{Repeat until done} \\
\bottomrule
\end{tabular}
\end{center}

\vspace{0.5cm}

\textbf{Advantages:}
\begin{itemize}
  \item Transparent reasoning (you see the LLM's thoughts)
  \item Robust error recovery (LLM can react to failures)
  \item Flexible—handles new tools and tasks without retraining
\end{itemize}

\docref{Yao et al., ReAct: Synergizing Reasoning and Acting in Language Models}

\end{frame}

% ============================================================
% SLIDE 6: Static vs Dynamic Agents (v1.2.x)
% ============================================================
\begin{frame}
\frametitle{Static vs Dynamic Agents in LangChain v1.2.x}

\begin{columns}
\column{0.5\textwidth}
\textbf{Static Agents}
\begin{itemize}
  \item Tools are fixed at setup
  \item Runs the same way every time
  \item Predictable, testable
  \item Example: calculator + web search
\end{itemize}

\vspace{0.5cm}

\textbf{Use when:}
\begin{itemize}
  \item You have a known set of tools
  \item Task is well-defined
\end{itemize}

\column{0.5\textwidth}
\textbf{Dynamic Agents}
\begin{itemize}
  \item Tools can change at runtime
  \item More flexible, adaptive
  \item Harder to debug
  \item Example: plugin-based system
\end{itemize}

\vspace{0.5cm}

\textbf{Use when:}
\begin{itemize}
  \item Tools are user-provided
  \item Extensible system needed
\end{itemize}
\end{columns}

\vspace{0.5cm}

\begin{tipbox}[LangChain v1.2.x]
\texttt{create\_react\_agent()} supports both—pass tools at initialization.
\end{tipbox}

\end{frame}

    llm=llm,
    tools=tools,
    prompt=None  # Uses default ReAct prompt

% SLIDE 10: Agent State and Message History
% ============================================================
\begin{frame}
\frametitle{Agent State and Message History}

The agent maintains a \textbf{message history}:

\vspace{0.3cm}

\begin{itemize}
  \item \textbf{User message:} Your question
  \item \textbf{AI message (Thought):} LLM's reasoning
  \item \textbf{Tool call:} Which tool to invoke + args
  \item \textbf{Tool message:} Result from the tool
  \item (repeat Thought → Tool call → Tool message)
  \item \textbf{Final AI message:} The answer
\end{itemize}

\vspace{0.5cm}

\textbf{Access state:}

\begin{lstlisting}
# Via AgentExecutor
result = executor.invoke({"input": "..."})
print(result["output"])  # Final answer

# Via intermediate steps
intermediate = result.get("intermediate_steps", [])
for step in intermediate:
    tool_name, tool_input, tool_output = step
    print(tool_name + ": " + tool_output)
\end{lstlisting}

\end{frame}

        if kind == "on_llm_start":
            print("LLM thinking...")
        elif kind == "on_tool_start":
            print(f"Calling {event['data']...}")
        elif kind == "on_tool_end":

        elif kind == "on_chain_end":

# Configure to interrupt before tool calls
agent_with_tools = create_react_agent(llm, tools)

def approval_fn(state):
    """Ask user before executing a tool."""
    tool_name = state["messages"][-1].tool_calls[0]["name"]
    approved = input(f"Approve {tool_name}? [y/n]: ")
    return approved.lower() == "y"

# Manually step through execution
state = {"input": "Delete all files..."}
for event in agent_with_tools.stream(state):
    if should_interrupt(event):
        if not approval_fn(event):
            print("Tool rejected by user")
            break
\end{lstlisting}

% SLIDE 18: Agents vs Chains
% ============================================================
\begin{frame}
\frametitle{When to Use Agents vs Chains}

\begin{center}
\begin{tabular}{lll}
\toprule
\textbf{Aspect} & \textbf{Chain} & \textbf{Agent} \\
\midrule
Flow & Fixed, linear & Dynamic, reasoning-based \\
Tool use & Hardcoded sequence & LLM decides \\
Flexibility & Low & High \\
Latency & Predictable & Variable \\
Complexity & Low & Higher \\
Debugging & Easier & Harder \\
\bottomrule
\end{tabular}
\end{center}

\vspace{0.5cm}

\textbf{Use Chains when:}
\begin{itemize}
  \item Flow is well-defined (e.g., ``fetch data → transform → summarize'')
  \item You need predictable latency
  \item Low error rates are critical
\end{itemize}

\textbf{Use Agents when:}
\begin{itemize}
  \item Problem requires reasoning (``what tool do I need?'')
  \item Tool sequence is unknown
  \item Adaptability matters
\end{itemize}

\end{frame}

% ============================================================
% SLIDE 19: Common Pitfalls
% ============================================================
\begin{frame}
\frametitle{Common Pitfalls and How to Avoid Them}

\begin{warningbox}[Pitfall 1: Infinite Loops]
Agent keeps calling the same tool over and over.

\textbf{Fix:} Set \texttt{max\_iterations}, use early stopping, improve tool descriptions.
\end{warningbox}

\begin{warningbox}[Pitfall 2: Poor Tool Descriptions]
Agent picks wrong tool because description is vague.

\textbf{Fix:} Write clear, precise tool docs with examples.
\end{warningbox}

\begin{warningbox}[Pitfall 3: Missing Error Handling]
Tool fails, agent halts or gives bad answer.

\textbf{Fix:} Wrap tools with try/catch, return informative error messages.
\end{warningbox}

\begin{warningbox}[Pitfall 4: Token Overflow]
Agent uses too many tokens and context window fills up.

\textbf{Fix:} Use message summarization, limit iterations, pick smaller LLM.
\end{warningbox}

\end{frame}

% SLIDE 20: Production-Ready Agent Pattern

\begin{frame}[fragile]
\frametitle{Production-Ready Agent Pattern}

A robust pattern for production systems:

\begin{lstlisting}
from langchain.agents import create_react_agent, AgentExecutor
from tenacity import retry, stop_after_attempt

class ProductionAgent:
    def __init__(self, llm, tools):
        self.agent = create_react_agent(llm, tools)
        self.executor = AgentExecutor(
            agent=self.agent,
            tools=tools,
            max_iterations=10,
            early_stopping_method="force",
            handle_parsing_errors=True
        )

    @retry(stop=stop_after_attempt(3))
    def run(self, query: str, timeout: int = 30) -> str:
        try:
            result = self.executor.invoke(
                {"input": query},
                config={"timeout": timeout}
            )
            return result["output"]
        except Exception as e:
            return f"Error: {str(e)[:100]}"
\end{lstlisting}

\textbf{Features:} Retries, timeouts, error handling, iteration limits.

\end{frame}

\begin{frame}[fragile]
\frametitle{Memory in Agents}

Agents have implicit memory via message history:

\begin{lstlisting}
# Each invoke call includes all prior messages
executor = AgentExecutor(agent=agent, tools=tools)

# First query
result1 = executor.invoke({"input": "My name is Alice"})

# Second query (with memory)
result2 = executor.invoke({"input": "What is my name?"})
# Agent remembers: Alice

\begin{frame}[fragile]
\frametitle{Tool Validation and Schemas}

Define strict schemas for tools:

\begin{lstlisting}
from pydantic import BaseModel, Field

class SearchInput(BaseModel):
    query: str = Field(description="Search query")
    max_results: int = Field(10, description="Max results")

@tool
def search(input: SearchInput) -> str:
    """Search the web with validation."""
    return f"Results for {input.query}"

# LLM respects the schema
tools = [search]
agent = create_react_agent(llm, tools)

# Agent passes correctly typed arguments
executor.invoke({"input": "Search for cats with 5 results"})
\end{lstlisting}

\textbf{Benefits:}
\begin{itemize}
  \item Type-safe tool calls
  \item LLM gets clear argument specs
  \item Runtime validation
\end{itemize}

% SLIDE 26: Tool Error Recovery
% ============================================================
\begin{frame}[fragile]
\frametitle{Tool Error Recovery}

Handle tool failures and let agent retry:

\begin{lstlisting}
from langchain_core.tools import tool

@tool
def unstable_api(query: str) -> str:
    """May fail sometimes."""
    import random
    if random.random() < 0.5:
        raise Exception("API temporarily down")
    return f"Result: {query}"

# Agent automatically retries on tool error
tools = [unstable_api]
executor = AgentExecutor(
    agent=agent,
    tools=tools,
    handle_parsing_errors=True
)

result = executor.invoke({"input": "Call API"})
# Agent may retry or use alternative approach
\end{lstlisting}

\textbf{Key:} Tools should return error messages, not throw exceptions.

\end{frame}

Key metrics to track in production:

\begin{itemize}

\begin{frame}[fragile]
\frametitle{Advanced: Conditional Logic via Tool Descriptions}

Guide agent behavior through tool descriptions:

\begin{lstlisting}
@tool
def search_internal(query: str) -> str:
    """Search only internal documents. Use for proprietary info."""
    return f"Internal result: {query}"

@tool
def search_web(query: str) -> str:
    """Search public web. Use for general knowledge."""
    return f"Web result: {query}"

# LLM reads descriptions and picks appropriate tool
tools = [search_internal, search_web]
agent = create_react_agent(llm, tools)

# For internal query, LLM chooses search_internal
# For general query, LLM chooses search_web
executor.invoke({"input": "Our Q4 revenue..."})
\end{lstlisting}

\textbf{Key:} Clear descriptions = better tool selection.

\end{frame}

% SLIDE 29: Tool Composition
% ============================================================
\begin{frame}[fragile]
\frametitle{Tool Composition: Tools that Call Tools}

Tools can be built from other tools:

\begin{lstlisting}
@tool
def compound_search(query: str) -> str:
    """Search web and email simultaneously."""
    web = search_web(query)
    email = search_email(query)
    return f"Web: {web}\nEmail: {email}"

# This tool is itself a tool that agent can call
tools = [compound_search, search_web, search_email]
agent = create_react_agent(llm, tools)
\end{lstlisting}

\textbf{Benefits:}
\begin{itemize}
  \item Reduce agent decision-making
  \item Pre-compose complex workflows
  \item Hide implementation details
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Summary: Section 05 - Agents}

\textbf{Core Concepts:}
\begin{enumerate}
  \item Agents are LLMs that reason and decide which tools to use
  \item ReAct pattern: Reason → Act → Observe → Repeat
  \item \texttt{create\_react\_agent()} builds agents on LangGraph
\end{enumerate}

\vspace{0.3cm}

\textbf{Implementation:}
\begin{enumerate}
  \item Define tools with clear descriptions
  \item Create agent with \texttt{create\_react\_agent(llm, tools)}
  \item Run via \texttt{AgentExecutor.invoke()} or stream
\end{enumerate}

\vspace{0.3cm}

\textbf{Production Readiness:}
\begin{enumerate}
  \item Set iteration limits, error handling, timeouts
  \item Use LangSmith for debugging and observability
  \item Test agents thoroughly (infinite loops, edge cases)
\end{enumerate}

\vspace{0.3cm}

\textbf{When to use agents:} Problems requiring reasoning and dynamic tool selection.

\end{frame}

% SLIDE 33: Suggested Answers
% ============================================================
\begin{frame}
\frametitle{Suggested Answers}

\textbf{Q1:} Chains have fixed, linear flow. Agents use LLM to decide tool sequence dynamically.

\vspace{0.2cm}

\textbf{Q2:} Agent thinks about a step, acts (calls a tool), observes the result, then repeats until done.

\vspace{0.2cm}

\textbf{Q3:} A \texttt{CompiledGraph} (LangGraph runnable state machine).

\vspace{0.2cm}

\textbf{Q4:} Poor descriptions cause agents to pick wrong tools or fail to recognize which tool to use.

\vspace{0.2cm}

\textbf{Q5:} (1) Infinite loops, (2) Poor tool descriptions, (3) Missing error handling, (4) Token overflow, (5) Hallucinated tools.

\vspace{0.2cm}

\textbf{Q6:} Set \texttt{LANGCHAIN\_TRACING\_V2=true} and \texttt{LANGCHAIN\_API\_KEY} environment variables.

\vspace{0.2cm}

\textbf{Q7:} Prevent infinite loops by halting execution after N iterations.

\vspace{0.2cm}

\textbf{Q8:} When the task requires reasoning, tool selection is unknown, or multiple tool paths are viable.
