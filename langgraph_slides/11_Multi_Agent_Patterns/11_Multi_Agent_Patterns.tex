\frametitle{Mental Model: Multi-Agent Systems}
\begin{conceptbox}[Core Idea]
\setlength{\itemsep}{1pt}
\begin{itemize}
\item Multiple specialized agents working together
\item Divide and conquer complex tasks
\item Agents handle different domains/skills
\item Orchestrator coordinates collaboration
\end{itemize}
\end{conceptbox}
\vspace{0.1cm}
\begin{tipbox}[Analogy]
Like a team: researcher, writer, reviewer, each expert.
\end{tipbox}

\begin{columns}[T]

\frametitle{Supervisor Pattern: Delegation}
\begin{conceptbox}[Architecture]
\setlength{\itemsep}{1pt}
\begin{itemize}
\item Supervisor LLM decides which worker
\item Workers execute specific tasks
\item Supervisor aggregates results
\item Hierarchical decision-making
\end{itemize}
\end{conceptbox}
\vspace{0.1cm}
\begin{tipbox}[Analogy]
Manager delegates to team; collects output.
\end{tipbox}
\docref{Supervisor Pattern}
\end{frame}

\begin{frame}
\frametitle{Swarm Pattern: Agent Handoffs}
\begin{conceptbox}[Key Idea]
\setlength{\itemsep}{1pt}
\begin{itemize}
\item Agents hand control to each other
\item No central supervisor
\item Peer-to-peer communication
\item Self-organizing workflow
\end{itemize}
\end{conceptbox}
\vspace{0.1cm}
\begin{tipbox}[When to Use]
\setlength{\itemsep}{1pt}
\begin{itemize}
\item Complex workflows, no clear hierarchy
\item Agents need flexibility
\item Natural conversation flow
\end{itemize}
\end{tipbox}
\docref{Swarm Pattern}
\end{frame}

\begin{frame}
\frametitle{Agent Handoffs}
\begin{tipbox}[Handoff Mechanism]
\texttt{def agent\_a(state):}\\
\texttt{if needs\_agent\_b:}\\
\texttt{return \{"next\_agent": "agent\_b"\}}
\end{tipbox}
\vspace{0.1cm}
\begin{tipbox}[Implementation]
\setlength{\itemsep}{1pt}
\begin{itemize}
\item Agent returns decision + context
\item Routing based on \texttt{next\_agent}
\item Seamless conversation transition
\end{itemize}
\end{tipbox}
\docref{Agent Handoffs}
\end{frame}

\end{itemize}
\end{conceptbox}
\vspace{0.1cm}
\begin{tipbox}[When to Use]
\setlength{\itemsep}{1pt}
\begin{itemize}
\item Very complex systems
\item Many specialized agents
\item Clear organizational structure
\end{itemize}
\end{tipbox}
\docref{Hierarchical Patterns}

\begin{frame}
\frametitle{Map-Reduce Pattern}
\begin{tipbox}[Dispatch to Workers]
\texttt{def map\_tasks(state):}\\
\texttt{tasks = split\_work(state["input"])}\\
\texttt{return [Send("worker", \{"task": t\})}\\
\texttt{for t in tasks]}
\end{tipbox}
\vspace{0.1cm}
\begin{tipbox}[Pattern Steps]
\setlength{\itemsep}{1pt}
\begin{itemize}
\item Map: split work across workers
\item Workers execute in parallel
\item Reduce: aggregate results
\end{itemize}
\end{tipbox}
\docref{Send, Map-Reduce}
\end{frame}

\begin{frame}
\frametitle{Shared vs Isolated State}
\begin{warningbox}[Shared State]
\setlength{\itemsep}{1pt}
\begin{itemize}
\item All agents see same state
\item Easy coordination
\item Risk of conflicts
\end{itemize}
\end{warningbox}
\vspace{0.1cm}
\begin{tipbox}[Isolated State]
\setlength{\itemsep}{1pt}
\begin{itemize}
\item Each agent private state
\item Explicit message passing
\item Clearer data flow
\end{itemize}
\end{tipbox}
\docref{State Management}
\end{frame}

\begin{frame}
\frametitle{Common Pitfalls}
\begin{warningbox}[Agent Loops]
Agents hand off back and forth endlessly.\\
\textbf{Fix:} Max iterations, clear termination.
\end{warningbox}
\vspace{0.08cm}
\begin{warningbox}[Unclear Responsibilities]
Agents overlap; confusion on roles.\\
\textbf{Fix:} Document specializations; test thoroughly.
\end{warningbox}
\vspace{0.08cm}
\begin{warningbox}[State Conflicts]
Multiple agents write same field.\\
\textbf{Fix:} Isolated state + explicit messaging.
\end{warningbox}
\end{frame}

\begin{frame}
\frametitle{Summary: Multi-Agent Cheat Sheet}
\begin{conceptbox}[Key Patterns]
\setlength{\itemsep}{1pt}
\begin{itemize}
\item Supervisor: central, 3-5 agents
\item Swarm: peer agents, 5-10
\item Hierarchical: multi-level, 10+
\item Map-Reduce: parallel execution
\item Shared message list communication
\item Max iterations prevent loops
\item Document roles explicitly
\item Test routing thoroughly
\end{itemize}
\end{conceptbox}
\end{frame}
